CREATE TABLE users (
  email VARCHAR(30),
  name VARCHAR(30),
  password VARCHAR(30),
  PRIMARY KEY (email)
);

CREATE TABLE usersFoodData (
  foodName VARCHAR(30),
  dayIn Date,
  dayOut Date,
  email VARCHAR(30),
  FOREIGN KEY (email) REFERENCES users(email)
);

CREATE TABLE foodReference (
  foodName VARCHAR(30),
  expiryDate int,
  PRIMARY KEY (foodName)
);
